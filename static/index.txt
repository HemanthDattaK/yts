// ── history ────────────────────────────────────────────────────────────────
function addToHistory(topic){
  chatHistory.unshift({topic, sessionId});
  renderHistory();
}
function renderHistory(){
  const el = document.getElementById("historyList");
  if(!chatHistory.length){ el.innerHTML = `<p style="font-size:.8rem;color:var(--muted)">No searches yet</p>`; return; }
  el.innerHTML = chatHistory.slice(0,12).map((h,i)=>`
    <div class="history-item ${h.sessionId===sessionId?'active':''}" title="${h.topic}">
      ${h.topic}
    </div>`).join("");
}

// ── append message helpers ─────────────────────────────────────────────────
function removeWelcome(){ const w=document.getElementById("welcome"); if(w) w.remove(); }

function appendMsg(role, html, id){
  removeWelcome();
  const msgs = document.getElementById("messages");
  const div = document.createElement("div");
  div.className = `msg ${role}`;
  if(id) div.id = id;
  const initials = role==="user" ? "U" : "V";
  div.innerHTML = `
    <div class="avatar">${initials}</div>
    <div class="bubble">${html}</div>`;
  msgs.appendChild(div);
  msgs.scrollTop = msgs.scrollHeight;
  return div;
}

function appendStatus(text){
  return appendMsg("ai status", `<div class="spinner"></div><span>${text}</span>`);
}

function updateStatus(el, text){
  if(el) el.querySelector(".bubble").innerHTML = `<div class="spinner"></div><span>${text}</span>`;
}

function removeEl(el){ if(el) el.remove(); }

// ── render markdown-ish text ───────────────────────────────────────────────
function renderMd(text){
  return text
    .replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;")
    .replace(/^## (.+)$/gm, '<div class="md-h2">$1</div>')
    .replace(/^### (.+)$/gm, '<div class="md-h3">$1</div>')
    .replace(/\*\*(.+?)\*\*/g, '<strong class="md-bold">$1</strong>')
    .replace(/\*(.+?)\*/g, '<em>$1</em>')
    .replace(/`(.+?)`/g, '<code>$1</code>')
    .replace(/^[-•] (.+)$/gm, '<li class="md-li">$1</li>')
    .replace(/(<li[\s\S]*?<\/li>)/g, '<ul>$1</ul>')
    .replace(/\n\n/g, '</p><p class="md-p">')
    .replace(/\n/g, '<br/>')
    .replace(/^(?!<)(.+)$/gm, '<p class="md-p">$1</p>');
}

// ── video cards block ──────────────────────────────────────────────────────
function renderVideosBlock(videos){
  videoCards = {};
  let html = `<div style="margin-bottom:6px;font-size:.82rem;font-weight:600;color:var(--muted)">
    Sources — ${videos.length} video${videos.length>1?'s':''} selected</div>
    <div class="videos-row">`;
  videos.forEach(v => {
    html += `
      <a class="video-card" id="vc-${v.video_id}" href="${v.url}" target="_blank">
        <img src="${v.thumbnail}" alt="" loading="lazy" onerror="this.style.display='none'"/>
        <div class="vc-info">
          <div class="vc-title">${v.title}</div>
          <div class="vc-meta">${v.channel}${v.views_fmt?' · '+v.views_fmt:''}${v.duration?' · '+v.duration:''}</div>
        </div>
        <span class="vc-badge processing" id="vcb-${v.video_id}">⏳</span>
      </a>`;
  });
  html += `</div>`;
  const msgEl = appendMsg("ai", html);
  videos.forEach(v => { videoCards[v.video_id] = document.getElementById(`vcb-${v.video_id}`); });
  return msgEl;
}

function setBadge(video_id, state){
  const el = videoCards[video_id];
  if(!el) return;
  el.className = `vc-badge ${state}`;
  el.textContent = state==="ready" ? "✓ Done" : state==="error" ? "✗ Failed" : "⏳";
}

// ── answer actions ─────────────────────────────────────────────────────────
function addAnswerActions(bubble, fullText){
  const div = document.createElement("div");
  div.className = "answer-actions";
  div.innerHTML = `
    <button class="action-btn" onclick="copyText(this, \`${fullText.replace(/`/g,"\\`")}\`)">📋 Copy</button>
    <button class="action-btn" onclick="downloadText(\`${fullText.replace(/`/g,"\\`")}\`)">⬇ Download</button>`;
  bubble.appendChild(div);
}

function copyText(btn, text){
  navigator.clipboard.writeText(text).then(()=>{
    btn.textContent = "✓ Copied!";
    setTimeout(()=>btn.textContent="📋 Copy", 2000);
  });
}
function downloadText(text){
  const a = document.createElement("a");
  a.href = URL.createObjectURL(new Blob([text],{type:"text/plain"}));
  a.download = "vidai_answer.txt";
  a.click();
}

// ── main send ──────────────────────────────────────────────────────────────
async function handleSend(){
  if(isStreaming) return;
  const topic    = document.getElementById("topicInput").value.trim();
  const question = document.getElementById("questionInput").value.trim();
  if(!topic) { document.getElementById("topicInput").focus(); return; }

  const numVideos = parseInt(document.getElementById("numVideos").value) || 2;
  const language  = document.getElementById("langSelect").value;

  // show user bubble
  const display = question || `Tell me about: ${topic}`;
  appendMsg("user", `<strong>${topic}</strong>${question ? '<br/><span style="opacity:.7;font-size:.88em">'+question+'</span>' : ""}`);

  document.getElementById("questionInput").value = "";
  document.getElementById("questionInput").style.height = "";
  setLoading(true);

  addToHistory(topic);

  let statusEl   = appendStatus("Searching YouTube...");
  let answerBubble = null;
  let answerText   = "";

  const body = JSON.stringify({ topic, question, num_videos: numVideos, language, session_id: sessionId });

  try {
    const resp = await fetch("/api/search", {
      method:"POST", headers:{"Content-Type":"application/json"}, body
    });
    await readSSE(resp, {
      status:        (d) => updateStatus(statusEl, d.message),
      videos:        (d) => { removeEl(statusEl); statusEl=null; renderVideosBlock(d.videos); },
      transcript_ready: (d) => setBadge(d.video_id, "ready"),
      transcript_error: (d) => setBadge(d.video_id, "error"),
      answer_start:  () => {
        if(statusEl){ removeEl(statusEl); statusEl=null; }
        const el = appendMsg("ai", '<span class="cursor"></span>');
        answerBubble = el.querySelector(".bubble");
      },
      token: (d) => {
        answerText += d.text;
        answerBubble.innerHTML = renderMd(answerText) + '<span class="cursor"></span>';
        document.getElementById("messages").scrollTop = 9999;
      },
      answer_done: (d) => {
        answerBubble.innerHTML = renderMd(d.full_answer);
        addAnswerActions(answerBubble, d.full_answer);
        document.getElementById("messages").scrollTop = 9999;
      },
      saved:  (d) => {},
      error:  (d) => {
        if(statusEl) removeEl(statusEl);
        appendMsg("ai status", `<span>❌ ${d.message}</span>`);
      },
    });
  } catch(e) {
    appendMsg("ai status", `<span>❌ Network error: ${e.message}</span>`);
  }

  setLoading(false);
}

// ── follow-up from sidebar / enter after first search ─────────────────────
async function followUp(question){
  if(isStreaming || !question.trim()) return;
  const language = document.getElementById("langSelect").value;

  appendMsg("user", question);
  setLoading(true);

  let answerBubble = null;
  let answerText   = "";
  let statusEl     = appendStatus("Thinking...");

  const body = JSON.stringify({ question, language, session_id: sessionId });

  try {
    const resp = await fetch("/api/followup", {
      method:"POST", headers:{"Content-Type":"application/json"}, body
    });
    await readSSE(resp, {
      answer_start: () => {
        removeEl(statusEl); statusEl=null;
        const el = appendMsg("ai", '<span class="cursor"></span>');
        answerBubble = el.querySelector(".bubble");
      },
      token: (d) => {
        answerText += d.text;
        answerBubble.innerHTML = renderMd(answerText) + '<span class="cursor"></span>';
        document.getElementById("messages").scrollTop = 9999;
      },
      answer_done: (d) => {
        answerBubble.innerHTML = renderMd(d.full_answer);
        addAnswerActions(answerBubble, d.full_answer);
        document.getElementById("messages").scrollTop = 9999;
      },
      error: (d) => {
        if(statusEl) removeEl(statusEl);
        appendMsg("ai status", `<span>❌ ${d.message}</span>`);
      },
    });
  } catch(e) {
    appendMsg("ai status", `<span>❌ ${e.message}</span>`);
  }

  setLoading(false);
}

// ── SSE reader ─────────────────────────────────────────────────────────────
async function readSSE(resp, handlers){
  const reader = resp.body.getReader();
  const dec    = new TextDecoder();
  let buf      = "";
  while(true){
    const {done,value} = await reader.read();
    if(done) break;
    buf += dec.decode(value, {stream:true});
    const lines = buf.split("\n");
    buf = lines.pop();
    for(const line of lines){
      if(!line.startsWith("data:")) continue;
      try{
        const obj = JSON.parse(line.slice(5).trim());
        const fn  = handlers[obj.event];
        if(fn) fn(obj);
      }catch{}
    }
  }
}
